#include<stdio.h>
void sort(int a[], int m)
{
    int i = 0;
    int j = 0;
    int tmp = 0;
    for (i = 0; i<m - 1; i++)
    {
        for (j = 0; j < m - i - 1; j++)
        {
            if (a[j]>a[j + 1])
            {
                tmp = a[j + 1];
                a[j + 1] = a[j];
                a[j] = tmp;
            }
        }
    }
}
void odd(int a[],int m) {

	for(i = 0;i < m;i++) {
		if(a[i] % 2 == 0) {
			n++;
		}
		else k++;
	}
	printf("偶数个数为：%d",n);
	printf("奇数个数为：%d",k); 
}
